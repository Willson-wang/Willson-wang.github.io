Canvas标签简单运用

发表于 2016-11-27

    canvas标签是h5中的重量级标签，该标签可以用于制作复杂or简单的图片，今天有时间尝试了一下，并做了一个小demo,写出来与大家分享；

先使用canvas标签绘制了一些简单的案例如矩形，原型，线段等；

//需要注意的一点是，canvas标签通过标签属性width与height设置的宽高
与通过css样式设置的width，height是有区别的，建议使用标签属性设置width与height

//获取canvas标签
var oCanvas =  document.getElementById("canv");
//获取canvas对象下的一个子对象，用于绘制各种二维图形，所有的绘制图形的方法都属于该对象
var ctx = oCanvas.getContext("2d");
//绘制矩形
ctx.fillStyle = "red";
ctx.fillRect(0,0,100,100);

//画线段，绘制三角形
//开始路径
ctx.beginPath();
//设置起点坐标
ctx.moveTo(200,200);
//设置三角形另外两个顶点
ctx.lineTo(300,300);
ctx.lineTo(200,300);
//回到三角形的起点
ctx.lineTo(200,200);
//绘制
ctx.stroke();
//结束路径
ctx.closePath();


//空心画圆
ctx.beginPath();
//arc()方法中的第一个参数为X坐标，第二个参数为Y坐标，
第三个参数为圆的半径，第四个参数为圆弧的起始点，第五个参数为圆弧的终止点
，第六个参数为绘制圆的方向，false为顺时针，true为逆时针
ctx.arc(500,200,100,Math.PI*2,false);
ctx.stroke();
ctx.closePath();

//实心圆
ctx.beginPath();
ctx.arc(500,200,100,90,false);
//fill()方法,用于设置图形是否填充的方法
ctx.fill();
ctx.closePath();

<script type="text/javascript">
    //获取canvas标签
    var oCanvas = document.querySelector("canvas");
    var ctx = oCanvas.getContext("2d");
    //获取清空画布内容标签
    var oBtn = document.querySelector("#clearBtn");
    //获取包含颜色选择的父标签
    var oChangeColor = document.querySelector("#changeColor");
    //获取显示颜色的标签
    var oB = document.querySelector("#show-color");
    //获取橡皮擦标签
    var oEraser = document.querySelector("#eraser");
    //设置一个橡皮擦开启开关
    var flag = true;


    //为canvas标签添加鼠标按下事件
    oCanvas.onmousedown = function (e){
        var e = e || event;
        var beginX = e.offsetX;
        var beginY = e.offsetY;
    //这种跟随效果要实现真正的跟随且不会出现断点的情况，最好的办法就是把移动过程中的每个坐标点都通过对象属性保存下来，
需要注意的是在mousedown事件中保存的坐标，需要在事件调用函数之前，因为没有保存该属性时，是取不到值的；
而在move事件中，要保存的旧的坐标一定要在事件函数之后，因为只有生成了此次的点，再把此次的坐标保存下来，下次调用就是拿的上次的点，这里注意down事件中保存属性的时候，
要+一个值or-一个值，目的是在于绘制一个点，如果不进行加or减的话，就会画不出东西，
而在move事件中，不能加原因是这里保存的是上一个点的坐标，如果我保存了值的话，
势必导致移动过程中的部分坐标无法描绘出来出现断点现象；
        this.oldPosition = {
            X: beginX +1,
            Y: beginY +1
        }

        //创建canvas对象，并设置起始路劲及起始点
        if(flag){
            picture(beginX,beginY);
        }else{
            clearFn(beginX,beginY);
        }


        //设置鼠标移动事件
        oCanvas.onmousemove = function (e){
            var e = e || event;
            var beginX = e.offsetX;
            var beginY = e.offsetY;
            //oCanvas.onmousedown();
            if(flag){
                picture(beginX,beginY);
            }else {
                clearFn(beginX,beginY);
            }

            //保存上一个起始点的位置是保证，下一个点的结束点就是上一个点的起始点
            this.oldPosition = {
                X: beginX,
                Y: beginY
            }

        }

    }

    oCanvas.onmouseup = function (){
        this.onmousemove = null;
    }

    //画点功能
    function picture(x,y){
        //设置线段的宽度
        ctx.lineWidth = 5;
        //设线段的形状
        ctx.lineCap = "round";
        ctx.beginPath();
        //起始点的坐标
        ctx.moveTo(x,y);
        ctx.lineTo(oCanvas.oldPosition.X,oCanvas.oldPosition.Y);
        ctx.stroke();
        ctx.closePath();
    }

    //清空画布内容
    oBtn.onclick = function (){
        //清除画布的原理是当canvas标签的宽高有改变时，整个canvas标签就会重置；
        ctx.clearRect(0,0,oCanvas.width,oCanvas.height);
    }

    //颜色选择功能
    oChangeColor.onclick = function (e){
        var e = e || event;
        if(e.target.type == "button"){
            ctx.strokeStyle = getComputedStyle(e.target,null)["background-color"];
            console.log(getComputedStyle(e.target,null)["background-color"]);
            //map数据可以通过get方法直接获取到某个key的value值；
            oB.innerHTML = colorMap.get(getComputedStyle(e.target,null)["background-color"]);
            console.log(ctx.strokeStyle);
            oB.style.color = ctx.strokeStyle;
            flag = true;
        }
    }

    //定义一个map数据类型用于保存颜色值及文字描述
    var colorMap = new Map (
        [
            ["rgb(0, 0, 0)","黑色"],
            ["rgb(255, 192, 203)","粉红色"],
            ["rgb(255, 0, 0)","红色"],
            ["rgb(255, 165, 0)","橙色"],
            ["rgb(139, 69, 19)","棕色"],
            ["rgb(138, 43, 226)","蓝色"]
        ]
    )

    //橡皮擦功能，核心思想是1.保存canvas当前的画面，2、裁剪，3、清空当前画布，4、还原场景
    function clearFn(x,y){
        //保存当前的画面
        ctx.save();
        //设置开始路径
        ctx.beginPath();
        //设置裁剪大小及裁剪
        ctx.arc(x,y,5,0,Math.PI*2,false);
        ctx.clip();
        //清空当前画布
        ctx.clearRect(0,0,oCanvas.width,oCanvas.height);
        //还原场景
        ctx.restore();
        ctx.closePath();
    }

    //为橡皮擦按钮添加点击功能
    oEraser.onclick = function (){
        flag = false;
    }


</script>
